// Ensure your filenames match this array exactly (Case Sensitive) const songVideos = ["song1.mp4", "song2.mp4", "song3.mp4", "song4.mp4", "song5.mp4"]; const goodJobVideo = "good-job.mp4"; const boxes = document.querySelectorAll('.box'); const overlay = document.getElementById('video-overlay'); const countOverlay = document.getElementById('countdown-overlay'); const videoPlayer = document.getElementById('game-video'); const blurPlayer = document.getElementById('video-blur-bg'); const startBtn = document.getElementById('start-btn'); const sfxShimmer = document.getElementById('sfx-shimmer'); const sfxStop = document.getElementById('sfx-stop'); function adjustSize(val) { document.documentElement.style.setProperty('--game-scale', val + 'vh'); } function toggleFullScreen() { const docEl = document.documentElement; // iPad specific fullscreen request const request = docEl.webkitRequestFullscreen || docEl.requestFullscreen; if (request) request.call(docEl); // "Unlock" the video player for sound on the first interaction videoPlayer.play().then(() => videoPlayer.pause()); } function runSelectionCycle() { startBtn.style.visibility = 'hidden'; let count = 0; let duration = 15 + Math.floor(Math.random() * 10); sfxShimmer.play(); const interval = setInterval(() => { boxes.forEach(b => b.classList.remove('shimmer')); boxes[count % 5].classList.add('shimmer'); // Adjusted for 5 boxes count++; if (count > duration) { clearInterval(interval); sfxShimmer.pause(); sfxShimmer.currentTime = 0; startCountdown(count % 5); } }, 130); } function startCountdown(videoIndex) { sfxStop.play(); countOverlay.style.display = 'flex'; let timer = 3; countOverlay.innerText = "READY?"; // Pre-load the selected song during the countdown videoPlayer.src = songVideos[videoIndex]; blurPlayer.src = songVideos[videoIndex]; videoPlayer.load(); blurPlayer.load(); const countInterval = setInterval(() => { if (timer > 0) { countOverlay.innerText = timer; timer--; } else if (timer === 0) { countOverlay.innerText = "GO!"; timer--; } else { clearInterval(countInterval); countOverlay.style.display = 'none'; playSequence(); } }, 800); } function playSequence() { overlay.style.display = 'block'; videoPlayer.play(); blurPlayer.play(); // Reset listener to avoid double-triggering videoPlayer.onended = null; videoPlayer.onended = () => { if (!videoPlayer.src.includes(goodJobVideo)) { videoPlayer.src = goodJobVideo; blurPlayer.src = goodJobVideo; videoPlayer.load(); blurPlayer.load(); videoPlayer.play(); blurPlayer.play(); } else { overlay.style.display = 'none'; startBtn.style.visibility = 'visible'; boxes.forEach(b => b.classList.remove('shimmer')); blurPlayer.pause(); } }; }
